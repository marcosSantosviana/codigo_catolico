<script>

function login() {
  if (document.getElementById('senha').value === "1234") {
    document.getElementById('login').style.display = 'none';
    document.getElementById('upload').style.display = 'block';
  } else {
    alert('Senha errada');
  }
}

function toggle(id) {
  const l = document.getElementById(id);
  l.style.display = l.style.display === 'block' ? 'none' : 'block';
}

function add() {
  const files = document.getElementById('file').files;
  const grupoId = document.getElementById('grupoSelect').value;

  if (!grupoId) {
    alert("⚠️ Escolha um grupo antes!");
    return;
  }

  const lista = document.getElementById(grupoId);

  for (let i = 0; i < files.length; i++) {
    const li = document.createElement('li');
    const link = document.createElement('a');

    link.textContent = files[i].name;
    link.href = URL.createObjectURL(files[i]);
    link.target = '_blank';

    li.appendChild(link);
    lista.appendChild(li);
  }
}

function atualizarSelect() {
  const select = document.getElementById('grupoSelect');
  select.innerHTML = '<option value="">Escolha o grupo</option>';

  const listas = document.querySelectorAll('.lista');
  listas.forEach(l => {
    const option = document.createElement('option');
    option.value = l.id;
    option.textContent = l.id.replace("lista_", "");
    select.appendChild(option);
  });
}

function criarGrupo() {
  const nome = document.getElementById('novoGrupo').value;
  if (!nome) {
    alert("Digite um nome!");
    return;
  }

  const id = "lista_" + nome.toLowerCase().replace(/\s+/g, '_');

  if (document.getElementById(id)) {
    alert("Grupo já existe!");
    return;
  }

  const container = document.getElementById('grupos');

  const div = document.createElement('div');
  div.className = 'grupo';

  const titulo = document.createElement('div');
  titulo.className = 'titulo';
  titulo.textContent = "🙏 " + nome;
  titulo.onclick = () => toggle(id);

  const ul = document.createElement('ul');
  ul.className = 'lista';
  ul.id = id;

  div.appendChild(titulo);
  div.appendChild(ul);

  container.appendChild(div);

  document.getElementById('novoGrupo').value = '';
  atualizarSelect();
}

/* FIGURINHAS CAINDO (FUNCIONA SEM INTERNET) */
function criarFigurinha() {
  const span = document.createElement("span");
  span.innerHTML = "🙏";

  span.className = "figurinha";
  span.style.left = Math.random() * 100 + "vw";
  span.style.fontSize = (20 + Math.random() * 25) + "px";
  span.style.animationDuration = (4 + Math.random() * 4) + "s";

  document.body.appendChild(span);

  setTimeout(() => {
    span.remove();
  }, 8000);
}

setInterval(criarFigurinha, 400);

// iniciar
atualizarSelect();
